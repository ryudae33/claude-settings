   270→    Sub SerialOpen()
   271→
   272→        Try
   273→            If Serial_RFID.IsOpen() = True Then
   274→                Serial_RFID.Close()
   275→            End If
   276→            Serial_RFID.PortName = PortNumber_RFID
   277→            Serial_RFID.ReceivedBytesThreshold = 50
   278→            Serial_RFID.BaudRate = 9600
   279→            Serial_RFID.DataBits = 8
   280→            Serial_RFID.Open()
   281→
   282→        Catch ex As Exception
   283→            MsgBox("Serial Open Module Fail " & PortNumber_RFID)
   284→        End Try
   285→
   286→        Try
   287→            If Serial_Scanner.IsOpen() = True Then
   288→                Serial_Scanner.Close()
   289→            End If
   290→            Serial_Scanner.PortName = PortNumber_SCANNER
   291→            Serial_Scanner.BaudRate = 9600
   292→            Serial_Scanner.DataBits = 8
   293→            Serial_Scanner.Open()
   294→
   295→        Catch ex As Exception
   296→
   297→            MsgBox("Serial Open Scanner Fail " & PortNumber_SCANNER)
   298→        End Try
   299→
   300→        Try
   301→            If Serial_Tool.IsOpen() = True Then
   302→                Serial_Tool.Close()
   303→            End If
   304→            Serial_Tool.PortName = PortNumber_TOOL
   305→            Serial_Tool.BaudRate = 9600
   306→            Serial_Tool.DataBits = 8
   307→            Serial_Tool.Open()
   308→            Serial_Tool.Write(Chr("7") & Chr("9") & Chr("7") & Chr("9") & Chr("2") & "00200001            " & Chr("0") & Chr("3"))

   311→            tmr_Tool.Start()
   312→
   313→        Catch ex As Exception

   887→    Private Function ASC2STR(ByVal str As String) As String
   888→
   889→        Dim One_Hex, Tmp_Hex, retVal As String
   890→        Dim i As Integer
   891→
   892→        retVal = ""
   893→        For i = 1 To Len(str)
   894→            One_Hex = Mid(str, i, 1)
   895→            If IsNumeric(One_Hex) Then
   896→                Tmp_Hex = Mid(str, i, 2)
   897→                i = i + 1
   898→            Else
   899→                Tmp_Hex = Mid(str, i, 4)
   900→                i = i + 3
   901→            End If
   902→            retVal = retVal & Chr("&H" & Tmp_Hex)
   903→        Next
   904→        Return retVal
   905→
