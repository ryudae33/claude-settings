    Private Sub Read_RFiD()

        If Serial_RFID.IsOpen = True Then

            'RFID Data length => 1A(26자리) 읽어오기 *앞 3자리 파레트 넘버 나머지 23자리 시리얼 넘버
            Serial_RFID.Write(Chr("&H02") & Chr("&H02") & Chr("&H00") & Chr("&H07") & Chr("&H05") & Chr("&H00") & Chr("&H00") &
                                          Chr("&H00") & Chr("&H1A") & Chr("&H03") & Chr("&HE8") & Chr("&H03"))
        End If

    End Sub

    Private Sub Write_RFID(ByVal str As String)

        Dim i As Integer

        Dim tmp1 As String = Chr("&H02") & Chr("&H02") & Chr("&H00") & Chr("&H1E") & Chr("&H06") & Chr("&H00") & Chr("&H00") &
                                                Chr("&H00") & Chr("&H17") & Chr("&H03") & Chr("&HE8")

        tmp1 = tmp1 & str & Chr("&H03")

        If Serial_RFID.IsOpen = True Then
            Serial_RFID.Write(tmp1)
        End If

    End Sub

    Sub SerialOpen()

        Try
            If Serial_RFID.IsOpen() = True Then
                Serial_RFID.Close()
            End If
            Serial_RFID.PortName = PortNumber_Rfid
            Serial_RFID.ReceivedBytesThreshold = 50
            Serial_RFID.BaudRate = 9600
            Serial_RFID.DataBits = 8
            Serial_RFID.Open()
        Catch ex As Exception
            MsgBox("Serial Open RFID Fail " & PortNumber_Rfid)
        End Try

        If PortNumber_Scanner = "NONE" Then
        Else
            Try
                If Serial_Scanner.IsOpen() = True Then
                    Serial_Scanner.Close()
                End If
                Serial_Scanner.PortName = PortNumber_Scanner
                Serial_Scanner.BaudRate = 9600
                Serial_Scanner.DataBits = 8

            End Try

        If PortNumber_Tool1 = "NONE" Then

                    Serial_Tool1.Close()
                    Serial_Tool1.DataBits = 8

    Private Sub Serial_Scanner_DataReceived(ByVal sender As Object, ByVal e As System.IO.Ports.SerialDataReceivedEventArgs) Handles Serial_Scanner.DataReceived

        Dim Incoming As String
        Incoming = Serial_Scanner.ReadLine()
        Dim ScanDAta As String = Mid$(Incoming, 1, Len(Incoming) - 1)
        Panel_Scanner_Input.Text = Mid$(Incoming, 1, Len(Incoming) - 1)

        '        첫자리 LH, RH
        '2~3자리 : 생산년도
        '4자리 : 생산월(A - 1월, B - 2월 .....)
        '5~6자리 : 생산일자
        '7~11자리 : 로트숫자
        '이후자리:   펌핑 작동력

        If wStep > 5 Then

            If (ScanDAta.Contains("LS") = True Or ScanDAta.Contains("RS") = True) And Panel_Pump_DEC.Text <> "NA" Then

                Panel_Pump_VAlue.Text = Panel_Scanner_Input.Text
                If CheckPumpDuplicated(Panel_Pump_VAlue.Text) = False Then
                    If Panel_Position.Text = "LH" Then
                        If Mid(Panel_Pump_VAlue.Text, 1, 2) = "LS" Then
                            Panel_Pump_DEC.Text = "OK"
                            Panel_Pump_DEC.BackColor = Color.Blue
                            DingDOng()
                        Else
                            Panel_Pump_DEC.Text = "NG"
                            Panel_Pump_DEC.BackColor = Color.Red
                            NG()

    Private Sub PlcRead()

        Dim iReturnCode As Integer              'Return code
        Dim szDeviceName As String = ""         'List data for 'DeviceName'
        Dim iNumberOfDeviceName As Integer = 0  'Data for 'DeviceSize'
        Dim sharrDeviceValue() As Short         'Data for 'DeviceValue'
        Dim szarrData() As String               'Array for 'Data'
        Dim iNumber As Integer                  'Loop counter
        Dim Arrysize As Integer = 11
        Dim j As Integer
        Dim i As Integer

        ReDim sharrDeviceValue(Arrysize - 1)
        ReDim szarrData(Arrysize - 1)

        'szDeviceName = "W0000" & vbLf &
        '                        "W0001" & vbLf &
        '                        "W0050" & vbLf &
        '                        "W0051" & vbLf &
        '                        "W0052" & vbLf &
        '                        "W0053" & vbLf &
        '                        "W0054" & vbLf &
        '                        "W0055" & vbLf &
        '                        "W0056" & vbLf &
        '                        "W0057"

        szDeviceName = "W" & Format(PlcInStartAddress, "0000") & vbLf &
                                "W" & Format(PlcInStartAddress + 1, "0000") & vbLf &
                                "W" & Format(PlcInStartAddress + 2, "0000") & vbLf &
                                "W" & Format(PlcOutStartAddress, "0000") & vbLf &
                                "W" & Format(PlcOutStartAddress + 1, "0000") & vbLf &
                                "W" & Format(PlcOutStartAddress + 2, "0000") & vbLf &
                                "W" & Format(PlcOutStartAddress + 3, "0000") & vbLf &
                                "W" & Format(PlcOutStartAddress + 4, "0000") & vbLf &
                                "W" & Format(PlcOutStartAddress + 5, "0000") & vbLf &
                                "W" & Format(PlcOutStartAddress + 6, "0000") & vbLf &
                                "W" & Format(PlcOutStartAddress + 7, "0000")
        Try
            iReturnCode = ACT_PLC1.ReadDeviceRandom2(szDeviceName, Arrysize, sharrDeviceValue(0))
        Catch exException As Exception
            'Exception processing
            MessageBox.Show(exException.Message, Name, MessageBoxButtons.OK, MessageBoxIcon.Error)
            Exit Sub
        End Try

        'When the ReadDeviceRandom2 method is succeeded, display the read data.
        If iReturnCode = 0 Then

            'Copy the read data to the 'lpszarrData'.
            For iNumber = 0 To iNumberOfDeviceName - 1
                szarrData(iNumber) = sharrDeviceValue(iNumber).ToString()
            Next iNumber
            For i = 0 To Arrysize - 1
                PLC_VALUE(i) = CInt(sharrDeviceValue(i))
            Next i

            PlcDw(PlcInStartAddress) = PLC_VALUE(0)
            PlcDw(PlcInStartAddress + 1) = PLC_VALUE(1)
            PlcDw(PlcInStartAddress + 2) = PLC_VALUE(2)

            PlcDw(PlcOutStartAddress) = PLC_VALUE(3)
            PlcDw(PlcOutStartAddress + 1) = PLC_VALUE(4)
            PlcDw(PlcOutStartAddress + 2) = PLC_VALUE(5)
            PlcDw(PlcOutStartAddress + 3) = PLC_VALUE(6)
            PlcDw(PlcOutStartAddress + 4) = PLC_VALUE(7)
            PlcDw(PlcOutStartAddress + 5) = PLC_VALUE(8)

            VALUE_IN_LABEL(0).Text = PlcDw(PlcInStartAddress)
            VALUE_IN_LABEL(1).Text = PlcDw(PlcInStartAddress + 1)
            VALUE_IN_LABEL(2).Text = PlcDw(PlcInStartAddress + 2)

            VALUE_OUT_LABEL(0).Text = PlcDw(PlcOutStartAddress)
            VALUE_OUT_LABEL(1).Text = PlcDw(PlcOutStartAddress + 1)
            VALUE_OUT_LABEL(2).Text = PlcDw(PlcOutStartAddress + 2)
            VALUE_OUT_LABEL(3).Text = PlcDw(PlcOutStartAddress + 3)
            VALUE_OUT_LABEL(4).Text = PlcDw(PlcOutStartAddress + 4)
            VALUE_OUT_LABEL(5).Text = PlcDw(PlcOutStartAddress + 5)

            For j = 0 To 2
                If VALUE_IN_LABEL(j).Text = "0" Then
                    VALUE_IN_LABEL(j).ForeColor = Color.Black
                    VALUE_IN_LABEL(j).BackColor = Color.White
                Else
                    VALUE_IN_LABEL(j).ForeColor = Color.White
                    VALUE_IN_LABEL(j).BackColor = Color.Black
                End If
            Next
            For j = 0 To 5
                If VALUE_OUT_LABEL(j).Text = "0" Then
                    VALUE_OUT_LABEL(j).ForeColor = Color.Black
                    VALUE_OUT_LABEL(j).BackColor = Color.White
                Else
                    VALUE_OUT_LABEL(j).ForeColor = Color.White
                    VALUE_OUT_LABEL(j).BackColor = Color.Black
                End If
            Next
            tError1 = "OK"

        Else

            tError1 = Dec2Hex(iReturnCode)

        End If

    End Sub

    Private Sub PLC_WRITE_1(ByVal strChr As String, ByVal StartArry As String, ByVal ArryMessage As String)

        Dim iReturnCode As Integer              'Return code
        Dim szDeviceName As String = ""         'List data for 'DeviceName'
        Dim sharrDeviceValue() As Short          'Data for 'DeviceValue'

        ReDim sharrDeviceValue(Len(ArryMessage))

        For i As Integer = 0 To Len(ArryMessage) - 1
            szDeviceName = szDeviceName & strChr & Format((Int(StartArry) + i), "0000")
            'If i <> Len(ArryMessage - 1) Then szDeviceName = szDeviceName + ControlChars.Lf
            If i < Len(ArryMessage) - 1 Then szDeviceName = szDeviceName & ControlChars.Lf
            sharrDeviceValue(i) = Mid(ArryMessage, i + 1, 1)
        Next

        Try

            iReturnCode = ACT_PLC1.WriteDeviceRandom2(szDeviceName, Len(ArryMessage), sharrDeviceValue(0))

        Catch exception As Exception

            'Exception processing
            'MessageBox.Show(exception.Message, Text, MessageBoxButtons.OK, MessageBoxIcon.Error)
            Flag_Connection_PLC1 = False
            Exit Sub

        End Try

        'The return code of the method is displayed by the hexadecimal.
        'txt_ReturnCode.Text = String.Format("0x{0:x8} [HEX]", iReturnCode)

    End Sub

    ' IO 조건 확인 함수 ("번호_ON/번호_OFF" 형식)
    Private Function CheckIOCondition(str As String) As Boolean
        If String.IsNullOrWhiteSpace(str) Then Return True

        Dim conditions() As String = str.Split("/"c)

        For Each cond As String In conditions
            cond = cond.Trim()
            If String.IsNullOrEmpty(cond) Then Continue For

            Dim parts() As String = cond.Split("_"c)
            If parts.Length <> 2 Then Return False

            Dim ioNumber As Integer
            If Not Integer.TryParse(parts(0), ioNumber) Then Return False

            Dim expectedValue As Integer
            If parts(1).ToUpper() = "ON" Then
                expectedValue = 1
            ElseIf parts(1).ToUpper() = "OFF" Then
                expectedValue = 0
            Else
                Return False
            End If

            If ioNumber < 0 OrElse ioNumber >= D_Value.Length Then Return False

            If D_Value(ioNumber) <> expectedValue Then
                Return False
            End If
        Next

        Return True
    End Function
    Private Function STR2ASC(ByVal str As String) As String

        Dim tmp As String = ""
        Dim i As Integer

        For i = 1 To Len(str)
            tmp = tmp & Hex(Asc(Mid(str, i, 1)))
        Next

        Return tmp

    End Function

    Private Function ASC2STR(ByVal str As String) As String

        Dim One_Hex, Tmp_Hex, i, retVal
        For i = 1 To Len(str)
            One_Hex = Mid(str, i, 1)
            If IsNumeric(One_Hex) Then
                Tmp_Hex = Mid(str, i, 2)
                i = i + 1
            Else
                Tmp_Hex = Mid(str, i, 4)
                i = i + 3
            End If
            retVal = retVal & Chr("&H" & Tmp_Hex)
        Next
        Return retVal

    End Function
