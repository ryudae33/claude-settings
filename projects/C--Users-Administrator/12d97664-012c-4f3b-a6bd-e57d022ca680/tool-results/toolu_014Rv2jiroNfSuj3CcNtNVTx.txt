   260→    Private Sub Read_RFiD()
   261→
   262→        If Serial_RFID.IsOpen = True Then
   263→
   264→            'RFID Data length => 1A(26자리) 읽어오기 *앞 3자리 파레트 넘버 나머지 23자리 시리얼 넘버
   265→            Serial_RFID.Write(Chr("&H02") & Chr("&H02") & Chr("&H00") & Chr("&H07") & Chr("&H05") & Chr("&H00") & Chr("&H00") &
   266→                                          Chr("&H00") & Chr("&H1A") & Chr("&H03") & Chr("&HE8") & Chr("&H03"))
   267→        End If
   268→
   269→    End Sub

   271→    Private Sub Write_RFID(ByVal str As String)
   272→
   273→        Dim i As Integer
   274→
   275→        Dim tmp1 As String = Chr("&H02") & Chr("&H02") & Chr("&H00") & Chr("&H1E") & Chr("&H06") & Chr("&H00") & Chr("&H00") &
   276→                                                Chr("&H00") & Chr("&H17") & Chr("&H03") & Chr("&HE8")
   277→
   278→        tmp1 = tmp1 & str & Chr("&H03")
   279→
   280→        If Serial_RFID.IsOpen = True Then
   281→            Serial_RFID.Write(tmp1)
   282→        End If

  1571→    ' IO 조건 확인 함수 ("번호_ON/번호_OFF" 형식)
  1572→    Private Function CheckIOCondition(str As String) As Boolean
  1573→        If String.IsNullOrWhiteSpace(str) Then Return True
  1574→
  1575→        Dim conditions() As String = str.Split("/"c)
  1576→
  1577→        For Each cond As String In conditions
  1578→            cond = cond.Trim()
  1579→            If String.IsNullOrEmpty(cond) Then Continue For
  1580→
  1581→            Dim parts() As String = cond.Split("_"c)
  1582→            If parts.Length <> 2 Then Return False
  1583→
  1584→            Dim ioNumber As Integer
  1585→            If Not Integer.TryParse(parts(0), ioNumber) Then Return False
  1586→
  1587→            Dim expectedValue As Integer
  1588→            If parts(1).ToUpper() = "ON" Then
  1589→                expectedValue = 1
  1590→            ElseIf parts(1).ToUpper() = "OFF" Then
  1591→                expectedValue = 0
  1592→            Else
  1593→                Return False
  1594→            End If
  1595→
  1596→            If ioNumber < 0 OrElse ioNumber >= D_Value.Length Then Return False
  1597→
  1598→            If D_Value(ioNumber) <> expectedValue Then
  1599→                Return False
  1600→            End If
  1601→        Next
  1602→
  1603→        Return True
  1604→    End Function
