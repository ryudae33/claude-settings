   687→
   688→        Dim One_Hex, Tmp_Hex, retVal As String
   689→        Dim i As Integer
   690→
   691→        retVal = ""

   693→            One_Hex = Mid(str, i, 1)
   694→            If IsNumeric(One_Hex) Then
   695→                Tmp_Hex = Mid(str, i, 2)
   696→                i = i + 1
   697→            Else
   698→                Tmp_Hex = Mid(str, i, 4)
   699→                i = i + 3
   700→            End If
   701→            retVal = retVal & Chr("&H" & Tmp_Hex)
