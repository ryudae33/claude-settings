   747→
   748→        Dim tmp As String = ""
   749→        Dim i As Integer
   750→
   751→        For i = 1 To Len(str)

   753→        Next
   754→
   755→        Return tmp

   759→    Private Function ASC2STR(ByVal str As String) As String
   760→
   761→        Dim One_Hex, Tmp_Hex, retVal As String
   762→        Dim i As Integer
   763→
   764→        retVal = ""
   765→        For i = 1 To Len(str)
   766→            One_Hex = Mid(str, i, 1)
   767→            If IsNumeric(One_Hex) Then
   768→                Tmp_Hex = Mid(str, i, 2)
   769→                i = i + 1
   770→            Else
   771→                Tmp_Hex = Mid(str, i, 4)
   772→                i = i + 3
   773→            End If
   774→            retVal = retVal & Chr("&H" & Tmp_Hex)
   775→        Next
   776→        Return retVal
   777→
   778→    End Function
