   527→

   871→    Private Function ASC2STR(ByVal str As String) As String
   872→
   873→        Dim One_Hex, Tmp_Hex, retVal As String
   874→        Dim i As Integer
   875→
   876→        retVal = ""
   877→        For i = 1 To Len(str)
   878→            One_Hex = Mid(str, i, 1)
   879→            If IsNumeric(One_Hex) Then
   880→                Tmp_Hex = Mid(str, i, 2)
   881→                i = i + 1
   882→            Else
   883→                Tmp_Hex = Mid(str, i, 4)
   884→                i = i + 3
   885→            End If
   886→            retVal = retVal & Chr("&H" & Tmp_Hex)
   887→        Next
   888→        Return retVal
   889→
   890→    End Function
