   755→    Private Function Laser2Angle(ByVal vLaser As Double) As Double
   756→
   757→        Dim tmp As Double = 0
   758→        For i As Integer = 0 To UBound(LASER) - 1
   759→            If vLaser >= LASER(i) And vLaser <= LASER(i + 1) Then
   760→                tmp = Calc(LASER(i), ANGLE(i), LASER(i + 1), ANGLE(i + 1), vLaser)
   761→            End If
   762→        Next i
   763→        Return tmp
   764→
   765→    End Function

   767→    Public Function Calc(ByVal aLaser As Double, ByVal aAngle As Double, ByVal bLaser As Double, ByVal bAngle As Double, ByVal val As Double) As Double
   768→
   769→        Dim t As Double
   770→        Dim tmp As Double = 0
   771→        Try
   772→            If aLaser = 0 And aAngle = 0 And bLaser = 0 And bAngle = 0 Then
   773→                tmp = 0
   774→            Else
   775→                t = (aAngle - bAngle) / (aLaser - bLaser)
   776→                tmp = (t * val) - (t * bLaser) + bAngle
   777→            End If
   778→        Catch ex As Exception
   779→        End Try
   780→        Return tmp
   781→
   782→    End Function

  1270→    Private Function ASC2STR(ByVal str As String) As String
  1271→
  1272→        Dim One_Hex, Tmp_Hex, retVal As String
  1273→        Dim i As Integer
  1274→
  1275→        retVal = ""
  1276→        For i = 1 To Len(str)
  1277→            One_Hex = Mid(str, i, 1)
  1278→            If IsNumeric(One_Hex) Then
  1279→                Tmp_Hex = Mid(str, i, 2)
  1280→                i = i + 1

  1284→            End If

  1892→
  1893→        Connection_Open_MDB()
  1894→        Dim Rs As New ADODB.Recordset
  1895→        Dim i As Integer
  1896→
  1897→        Rs.CursorLocation = ADODB.CursorLocationEnum.adUseClient
  1898→        Rs.CursorType = ADODB.CursorTypeEnum.adOpenStatic
  1899→        Rs.LockType = ADODB.LockTypeEnum.adLockOptimistic
  1900→
  1901→        Rs.Open("SELECT * FROM Table_Angle ORDER BY LEN(LASER_LENGTH),LASER_LENGTH", Mdb_Connect)
  1902→
  1903→        ReDim ANGLE(Rs.RecordCount)
  1904→        ReDim LASER(Rs.RecordCount)
  1905→
  1906→        If Rs.RecordCount = 0 Then
  1907→
  1908→        ElseIf Rs.RecordCount = 1 Then
  1909→
  1910→            ANGLE(0) = Rs.Fields("CONVERT_ANGLE").Value
  1911→            LASER(0) = Rs.Fields("LASER_LENGTH").Value
  1912→
  1913→        ElseIf Rs.RecordCount > 1 Then
  1914→
  1915→            i = 0

  1917→
