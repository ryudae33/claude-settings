   261→    Private Sub Read_RFiD()
   262→
   263→        If Serial_RFID.IsOpen = True Then
   264→
   265→            'RFID Data length => 1A(26자리) 읽어오기 *앞 3자리 파레트 넘버 나머지 23자리 시리얼 넘버
   266→            Serial_RFID.Write(Chr("&H02") & Chr("&H02") & Chr("&H00") & Chr("&H07") & Chr("&H05") & Chr("&H00") & Chr("&H00") &
   267→                                          Chr("&H00") & Chr("&H1A") & Chr("&H03") & Chr("&HE8") & Chr("&H03"))
   268→        End If
   269→
   270→    End Sub

   272→    Private Sub Write_RFID(ByVal str As String)
   273→
   274→        Dim i As Integer
   275→
   276→        Dim tmp1 As String = Chr("&H02") & Chr("&H02") & Chr("&H00") & Chr("&H1E") & Chr("&H06") & Chr("&H00") & Chr("&H00") &
   277→                                                Chr("&H00") & Chr("&H17") & Chr("&H03") & Chr("&HE8")
   278→
   279→        tmp1 = tmp1 & str & Chr("&H03")
   280→
   281→        If Serial_RFID.IsOpen = True Then
   282→            Serial_RFID.Write(tmp1)
   283→        End If

   790→
   791→        Dim iReturnCode As Integer              'Return code
   792→        Dim szDeviceName As String = ""         'List data for 'DeviceName'
   793→        Dim iNumberOfDeviceName As Integer = 0  'Data for 'DeviceSize'
   794→        Dim sharrDeviceValue() As Short         'Data for 'DeviceValue'

  1572→    ' IO 조건 확인 함수 ("번호_ON/번호_OFF" 형식)
  1573→    Private Function CheckIOCondition(str As String) As Boolean
  1574→        If String.IsNullOrWhiteSpace(str) Then Return True
  1575→
  1576→        Dim conditions() As String = str.Split("/"c)
  1577→
  1578→        For Each cond As String In conditions
  1579→            cond = cond.Trim()
  1580→            If String.IsNullOrEmpty(cond) Then Continue For
  1581→
  1582→            Dim parts() As String = cond.Split("_"c)
  1583→            If parts.Length <> 2 Then Return False
  1584→
  1585→            Dim ioNumber As Integer
  1586→            If Not Integer.TryParse(parts(0), ioNumber) Then Return False
  1587→
  1588→            Dim expectedValue As Integer
  1589→            If parts(1).ToUpper() = "ON" Then
  1590→                expectedValue = 1
  1591→            ElseIf parts(1).ToUpper() = "OFF" Then
  1592→                expectedValue = 0
  1593→            Else
  1594→                Return False
  1595→            End If
  1596→
  1597→            If ioNumber < 0 OrElse ioNumber >= D_Value.Length Then Return False
  1598→
  1599→            If D_Value(ioNumber) <> expectedValue Then
  1600→                Return False
  1601→            End If
  1602→        Next
  1603→
  1604→        Return True
