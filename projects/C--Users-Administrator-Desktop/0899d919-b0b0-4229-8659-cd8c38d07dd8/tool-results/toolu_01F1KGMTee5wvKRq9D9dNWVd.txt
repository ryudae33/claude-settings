   204→        private void LoadItems()
   205→        {
   206→            _items.Clear();
   207→            string filePath = GetItemsFilePath();
   208→
   209→            if (File.Exists(filePath))
   210→            {
   211→                try
   212→                {
   213→                    foreach (var line in File.ReadAllLines(filePath, System.Text.Encoding.UTF8))
   214→                    {
   215→                        string trimmed = line.Trim();
   216→                        if (string.IsNullOrEmpty(trimmed) || trimmed.StartsWith("#")) continue;
   217→                        string[] parts = trimmed.Split('\t');
   218→                        if (parts.Length >= 4)
   219→                            _items.Add(new[] { parts[0], parts[1], parts[2], parts[3] });
   220→                    }
   221→                }
   222→                catch (Exception ex)
   223→                {
   224→                    Logger.Error($"[일상점검] 항목 로드 실패: {ex.Message}");
   225→                }
   226→            }
   227→
   228→            if (_items.Count == 0)
   229→            {
   230→                foreach (var item in DEFAULT_ITEMS)
   231→                    _items.Add((string[])item.Clone());
   232→                SaveItems();
   233→            }
   234→        }
   235→
   236→        private void SaveItems()
   237→        {
   238→            try
   239→            {
   240→                if (!Directory.Exists(_basePath))
   241→                    Directory.CreateDirectory(_basePath);
   242→
   243→                string filePath = GetItemsFilePath();
   244→                using (var sw = new StreamWriter(filePath, false, System.Text.Encoding.UTF8))
   245→                {
   246→                    sw.WriteLine("# 점검항목 설정 (탭 구분: 점검항목\\t점검기준\\t점검방법\\t구분)");
   247→                    foreach (var item in _items)
   248→                        sw.WriteLine($"{item[0]}\t{item[1]}\t{item[2]}\t{item[3]}");
   249→                }
   250→            }
   251→            catch (Exception ex)
   252→            {
   253→                Logger.Error($"[일상점검] 항목 저장 실패: {ex.Message}");
   254→            }
   255→        }

   424→        private void Dgv_CellClick(object sender, DataGridViewCellEventArgs e)
   425→        {
   426→            if (_loading || e.RowIndex < 0 || e.ColumnIndex < FIXED_COL_COUNT) return;
   427→
   428→            string colName = dgv.Columns[e.ColumnIndex].Name;
   429→            if (!colName.StartsWith("colDay")) return;
   430→
   431→            // colDay{d}_{M|A|E} 파싱
   432→            string suffix = colName.Substring(6); // "d_M" or "dd_M"
   433→            string[] parts = suffix.Split('_');
   434→            if (parts.Length != 2) return;
   435→
   436→            int day;
   437→            if (!int.TryParse(parts[0], out day)) return;
   438→            string shiftKey = parts[1];
   439→
   440→            int daysInMonth = DateTime.DaysInMonth(_currentMonth.Year, _currentMonth.Month);
   441→            if (day > daysInMonth) return;
   442→
   443→            var cell = dgv.Rows[e.RowIndex].Cells[e.ColumnIndex];
   444→            string current = cell.Value?.ToString()?.Trim() ?? "";
   445→
   446→            // 순환: "" → O → X → ""
   447→            string next;
   448→            if (current == "") next = "O";
   449→            else if (current == "O") next = "X";
   450→            else next = "";
   451→
   452→            cell.Value = next;
   453→            ApplyCellColor(cell, next, colName);
   454→
   455→            // 교대 이름 변환
   456→            int shiftIdx = Array.IndexOf(SHIFT_KEYS, shiftKey);
   457→            string shiftName = shiftIdx >= 0 ? SHIFT_FULL[shiftIdx] : shiftKey;
   458→
   459→            string itemName = dgv.Rows[e.RowIndex].Cells["colItem"].Value?.ToString() ?? "";
   460→            LogCheckAction(e.RowIndex + 1, itemName, day, shiftName, current, next);
   461→
   462→            SaveData();
   463→        }

   465→        private void ApplyCellColor(DataGridViewCell cell, string value, string colName = "")
   466→        {
   467→            if (value == "O")
   468→            {
   469→                cell.Style.ForeColor = Color.LimeGreen;
   470→                cell.Style.BackColor = Color.FromArgb(30, 60, 30);
   471→            }
   472→            else if (value == "X")
   473→            {
   474→                cell.Style.ForeColor = Color.Red;
   475→                cell.Style.BackColor = Color.FromArgb(70, 25, 25);
   476→            }
   477→            else
   478→            {
   479→                cell.Style.ForeColor = ThemeManager.DarkTheme.TextPrimary;
   480→                // 교대별 기본 배경색 복원
   481→                if (colName.EndsWith("_M"))
   482→                    cell.Style.BackColor = Color.FromArgb(38, 38, 48);
   483→                else if (colName.EndsWith("_A"))
   484→                    cell.Style.BackColor = Color.FromArgb(38, 44, 38);
   485→                else if (colName.EndsWith("_E"))
   486→                    cell.Style.BackColor = Color.FromArgb(44, 38, 38);
   487→                else
   488→                    cell.Style.BackColor = ThemeManager.DarkTheme.BackgroundMedium;
   489→            }
   490→        }

   515→
   516→            int count = Math.Min(_items.Count, 15);  // 최대 15개 표시

   541→                lbl.ForeColor = Color.FromArgb(180, 180, 180);
   542→                lbl.Tag = i;
   543→                lbl.Cursor = Cursors.Hand;
   544→                lbl.Click += PhotoThumb_Click;
   545→                panelPhotoScroll.Controls.Add(lbl);
   546→
   547→                // 썸네일 PictureBox

   549→                pb.Location = new Point(px, py + lblH);
   550→                pb.Size = new Size(itemW, thumbH);

   826→        /// <summary>
   827→        /// 파일에서 기존 데이터 로드 (다른 범위 데이터 보존용)
   828→        /// </summary>
   829→        private Dictionary<string, Dictionary<string, string>> LoadExistingData()
   830→        {
   831→            var result = new Dictionary<string, Dictionary<string, string>>();
   832→            string filePath = GetDataFilePath();
   833→            if (!File.Exists(filePath)) return result;
   834→
   835→            try
   836→            {
   837→                string currentSection = "";
   838→                foreach (var line in File.ReadAllLines(filePath, System.Text.Encoding.UTF8))
   839→                {
   840→                    string trimmed = line.Trim();
   841→                    if (string.IsNullOrEmpty(trimmed)) continue;
   842→
   843→                    if (trimmed.StartsWith("[") && trimmed.EndsWith("]"))
   844→                    {
   845→                        currentSection = trimmed.Substring(1, trimmed.Length - 2);
   846→                        continue;
   847→                    }
   848→
   849→                    int eqIdx = trimmed.IndexOf('=');
   850→                    if (eqIdx < 0) continue;
   851→                    string key = trimmed.Substring(0, eqIdx);
   852→                    string val = trimmed.Substring(eqIdx + 1);
   853→
   854→                    if (currentSection == "Header") continue;
   855→
   856→                    if (currentSection.StartsWith("Item"))
   857→                    {
   858→                        if (!result.ContainsKey(currentSection))
   859→                            result[currentSection] = new Dictionary<string, string>();
   860→                        result[currentSection][key] = val;
   861→                    }
   862→                }
   863→            }
   864→            catch { }
   865→            return result;
   866→        }
